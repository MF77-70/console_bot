import pickle

# --- –ö–ª–∞—Å AddressBook ---
class AddressBook(UserDict):
    """–ö–ª–∞—Å –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è –∑–∞–ø–∏—Å—ñ–≤ (Record) —Ç–∞ –∫–µ—Ä—É–≤–∞–Ω–Ω—è –Ω–∏–º–∏, –∑ —Ñ—É–Ω–∫—Ü—ñ–æ–Ω–∞–ª–æ–º –∑–±–µ—Ä–µ–∂–µ–Ω–Ω—è/–∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è."""
    
    def add_record(self, record):
        self.data[record.name.value] = record

    def find(self, name):
        return self.data.get(name)

    def delete(self, name):
        if name in self.data:
            del self.data[name]
        else:
            raise KeyError(f"–ö–æ–Ω—Ç–∞–∫—Ç –∑ —ñ–º'—è–º '{name}' –Ω–µ –∑–Ω–∞–π–¥–µ–Ω–æ.")

    # üü° –ú–µ—Ç–æ–¥ –¥–ª—è –∑–±–µ—Ä–µ–∂–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö —É —Ñ–∞–π–ª
    def save_to_file(self, filename):
        """–ó–±–µ—Ä—ñ–≥–∞—î AddressBook —É —Ñ–∞–π–ª –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é pickle."""
        with open(filename, "wb") as f:
            pickle.dump(self, f)  # üî¥ –°–∞–º–∞ –æ–ø–µ—Ä–∞—Ü—ñ—è –∑–∞–ø–∏—Å—É —É —Ñ–∞–π–ª

    # üü° –ú–µ—Ç–æ–¥ –¥–ª—è –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö —ñ–∑ —Ñ–∞–π–ª—É
    @classmethod
    def load_from_file(cls, filename):
        """–ó–∞–≤–∞–Ω—Ç–∞–∂—É—î AddressBook —ñ–∑ —Ñ–∞–π–ª—É."""
        try:
            with open(filename, "rb") as f:
                return pickle.load(f)  # üî¥ –û–ø–µ—Ä–∞—Ü—ñ—è —á–∏—Ç–∞–Ω–Ω—è –∑ —Ñ–∞–π–ª—É
        except FileNotFoundError:
            return cls()
        except Exception:
            return cls()
